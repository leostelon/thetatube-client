@public
collection Video {
  id: string; // theta videoID
  publicKey?: PublicKey;
  name: string;
  description: string;
  creator: User;
  // length: number; // Add length to video
  drm: boolean;
  nft_address?: string;
  timestamp: number;
  views: number;
  thumbnail: string;

  @index(drm, timestamp);

  constructor (id: string, name: string, description: string, creator: User, drm: boolean, nft_address: string, thumbnail: string) {
    this.id = id;
    this.name = name;
    this.description = description;
    this.creator = creator;
    this.drm = drm;
    this.nft_address = nft_address;
    this.timestamp = Date.now();
    this.views = 0;
    this.thumbnail = thumbnail;
  }

  incrementView() {
    this.views = this.views + 1;
  }
}
